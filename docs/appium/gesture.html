 70        """

116    def _set_boundaries_and_scrollable_area(self) -> None:
117        """
118        Calculate scrolling boundaries and scrollable area based on crop factors.
119        """
120        self.bounds = {
121            "upper": int(self.viewport_height * self.crop_factors["upper_cf"]),
122            "lower": int(self.viewport_height * self.crop_factors["lower_cf"]),
123            "left": int(self.viewport_width * self.crop_factors["left_cf"]),
124            "right": int(self.viewport_width * self.crop_factors["right_cf"]),
125        }
126
127        self.scrollable_area = {
128            "x": self.bounds["right"] - self.bounds["left"],
129            "y": self.bounds["lower"] - self.bounds["upper"],
130        }

150    def _calculate_element_points(self, element) -> dict[str, tuple[int, int]]:
151        """
152        Calculate various points on an element.
153
154        Args:
155            element: The WebElement to calculate points for.
156
157        Returns:
158            A dictionary containing coordinates of various points on the element.
159        """
160        try:
161            x, y = element.location["x"], element.location["y"]
162            width, height = element.size["width"], element.size["height"]
163
164            if width <= 0 or height <= 0:
165                raise ElementInteractionError("Invalid element dimensions")
166
167            return {
168                "top_left": (x, y),
169                "top_mid": (x + width // 2, y),
170                "top_right": (x + width, y),
171                "left_mid": (x, y + height // 2),
172                "mid": (x + width // 2, y + height // 2),
173                "right_mid": (x + width, y + height // 2),
174                "bottom_left": (x, y + height),
175                "bottom_mid": (x + width // 2, y + height),
176                "bottom_right": (x + width, y + height),
177            }
178        except Exception as e:
179            msg = f"Failed to calculate element points: {str(e)}"
180            logger.error(msg)
181            raise ElementInteractionError(msg) from e

286    def swipe_on_element(
287        self, locator_method: AppiumBy, locator_value: str, direction: Direction
288    ) -> None:
289        """
290        Swipe on a specific element in the given direction.
291
292        Args:
293            locator_method: The method to locate the element (e.g., AppiumBy.XPATH).
294            locator_value: The value to use with the locator method.
295            direction: The direction to swipe (UP, DOWN, LEFT, or RIGHT).
296        """
297        try:
298            action = self._create_action()
299            element = self.driver.find_element(by=locator_method, value=locator_value)
300            element_points = self._calculate_element_points(element)
301
302            match direction:
303                case Direction.UP:
304                    self.perform_navigation_on_element(
305                        action, element_points["bottom_mid"], element_points["top_mid"]
306                    )
307                case Direction.DOWN:
308                    self.perform_navigation_on_element(
309                        action, element_points["top_mid"], element_points["bottom_mid"]
310                    )
311                case Direction.RIGHT:
312                    self.perform_navigation_on_element(
313                        action, element_points["left_mid"], element_points["right_mid"]
314                    )
315                case Direction.LEFT:
316                    self.perform_navigation_on_element(
317                        action, element_points["right_mid"], element_points["left_mid"]
318                    )
319        except Exception as e:
320            msg = f"Failed to swipe on element: {str(e)}"
321            logger.error(msg)
322            raise SwipeError(msg) from e

324    def swipe_element_into_view(
325        self, locator_method: AppiumBy, locator_value: str, direction: SeekDirection
326    ) -> None:
327        """
328        Swipe to bring an element into view.
329        The multipliers in `swipe_actions` method scale the percentage factor for swipe for the partial_percentage argument.
330
331        Args:
332            locator_method: The method to locate the element (e.g., AppiumBy.XPATH).
333            locator_value: The value to use with the locator method.
334            direction: The direction to search for the element (UP, DOWN, LEFT, or RIGHT).
335        """
336        try:
337            action = self._create_action()
338
339            if self._probe_for_element(locator_method, locator_value):
340                element_x, element_y = self._retrieve_element_location(
341                    locator_method, locator_value
342                )
343
344                if direction in [SeekDirection.UP, SeekDirection.DOWN]:
345                    self._swipe_element_into_view_vertical(action, element_y, direction)
346                elif direction in [SeekDirection.LEFT, SeekDirection.RIGHT]:
347                    self._swipe_element_into_view_horizontal(
348                        action, element_x, direction
349                    )
350            else:
351                swipe_actions = {
352                    SeekDirection.UP: lambda: self.perform_navigation_partial_y(
353                        action,
354                        self.bounds["upper"],
355                        self.bounds["lower"],
356                        self.scrollable_area["y"] * -0.4,
357                    ),
358                    SeekDirection.DOWN: lambda: self.perform_navigation_partial_y(
359                        action,
360                        self.bounds["lower"],
361                        self.bounds["upper"],
362                        self.scrollable_area["y"] * 0.4,
363                    ),
364                    SeekDirection.LEFT: lambda: self.perform_navigation_partial_x(
365                        action,
366                        self.bounds["left"],
367                        self.bounds["right"],
368                        self.scrollable_area["x"] * -0.2,
369                    ),
370                    SeekDirection.RIGHT: lambda: self.perform_navigation_partial_x(
371                        action,
372                        self.bounds["right"],
373                        self.bounds["left"],
374                        self.scrollable_area["x"] * 0.2,
375                    ),
376                }
377
378                for _ in range(self.probe_attempts):
379                    if self._probe_for_element(locator_method, locator_value):
380                        return
381                    swipe_actions[direction]()
382
383                raise ElementNotInViewError(
384                    f"Element not found after {self.probe_attempts} attempts"
385                )
386
387        except ElementNotInViewError:
388            raise
389        except Exception as e:
390            msg = f"Failed to swipe element into view: {str(e)}"
391            logger.error(msg)
392            raise SwipeError(msg) from e

394    def _swipe_element_into_view_vertical(
395        self, action: ActionChains, element_y: int, direction: SeekDirection
396    ) -> None:
397        """
398        Perform vertical swipes to bring an element into view.
399
400        Args:
401            action: The ActionChains object to use for swiping.
402            element_y: The y-coordinate of the element to bring into view.
403            The direction to search for the element (UP or DOWN).
404        """
405        try:
406            distance_to_element = element_y - self.bounds["lower"]
407            actions_total = distance_to_element / self.scrollable_area["y"]
408            actions_complete = int(distance_to_element // self.scrollable_area["y"])
409            actions_partial = int(
410                self.scrollable_area["y"] * (actions_total - actions_complete)
411            )
412
413            if direction == SeekDirection.UP:
414                start, end = self.bounds["upper"], self.bounds["lower"]
415            else:
416                start, end = self.bounds["lower"], self.bounds["upper"]
417
418            if actions_total > 1:
419                self.perform_navigation_full_y(action, start, end, actions_complete)
420            if actions_partial > 50:
421                self.perform_navigation_partial_y(action, start, end, actions_partial)
422
423        except Exception as e:
424            msg = f"Failed to swipe element into view vertically: {str(e)}"
425            logger.error(msg)
426            raise SwipeError(msg) from e

596    def _perform_swipe(
597        self, action: ActionChains, start: tuple[int, int], end: tuple[int, int]
598    ) -> None:
599        """
600        Perform a swipe action from start to end coordinates.
601
602        Args:
603            action: The ActionChains object to use for swiping.
604            start: The starting coordinates (x, y).
605            end: The ending coordinates (x, y).
606        """
607        try:
608            action.w3c_actions.pointer_action.move_to_location(*start)
609            action.w3c_actions.pointer_action.pointer_down()
610            action.w3c_actions.pointer_action.move_to_location(*end)
611            action.w3c_actions.pointer_action.pause(0.5)
612            action.w3c_actions.pointer_action.release()
613        except Exception as e:
614            msg = f"Failed to perform swipe action: {str(e)}"
615            logger.error(msg)
616            raise SwipeError(msg) from e

656        self,
657        locator_method: AppiumBy,
658        locator_value: str,
659        duration: float = 0.1,
660        iterations: int = 1,
661    ) -> None:
662        """
663        Helper method to tap on a specified element a given number of times.

680                action.w3c_actions.pointer_action.pause(duration)
681                action.w3c_actions.pointer_action.release()

808        locator_value: str,
809        direction: Direction,
810        duration: float = 0.5,
811    ) -> bool:
812        """
813        Performs a pinch-to-zoom gesture on the specified element.
814
815        Args:
816            locator_method (AppiumBy): The method to locate the element (e.g., ID, XPATH)
817            locator_value (str): The locator value to find the element
818            direction (Literal["in", "out"]): Direction of zoom - "in" for zoom in, "out" for zoom out
819            duration (float, optional): Duration of the zoom animation in seconds. Defaults to 0.5
820
821        Returns:
822            bool: True if zoom action was successful, False otherwise
823
824        Raises:
825            NoSuchElementException: If the target element cannot be found
826            ElementNotInteractableException: If the element is not interactable
827        """
828        try:
829            action = ActionChains(self.driver)
830
831            element = self.driver.find_element(by=locator_method, value=locator_value)
832            if not element.is_displayed() or not element.is_enabled():
833                raise ElementNotInteractableException("Element is not interactable")
834
835            element_points = self._calculate_element_points(element)
836
837            if direction == Direction.IN:
838                pointer_a_start = element_points["mid"]
839                pointer_b_start = element_points["mid"]
840                pointer_a_end = self.bounds["upper"]
841                pointer_b_end = self.bounds["lower"]
842            elif direction == Direction.OUT:
843                pointer_a_start = self.bounds["upper"]
844                pointer_b_start = self.bounds["lower"]
845                pointer_a_end = element_points["mid"]
846                pointer_b_end = element_points["mid"]
847
848            pointer_a = action.w3c_actions.add_pointer_input("touch", "pointer_a")
849            pointer_b = action.w3c_actions.add_pointer_input("touch", "pointer_b")
850
851            pointer_a.create_pointer_move(**pointer_a_start)
852            pointer_b.create_pointer_move(**pointer_b_start)
853
854            pointer_a.create_pointer_down()
855            pointer_b.create_pointer_down()
856
857            pointer_a.create_pause(duration * 0.1)
858            pointer_b.create_pause(duration * 0.1)
859
860            pointer_a.create_pointer_move(duration=duration, **pointer_a_end)
861            pointer_b.create_pointer_move(duration=duration, **pointer_b_end)
862
863            pointer_a.create_pointer_up()
864            pointer_b.create_pointer_up()
865
866            action.perform()
867            return True
868
869        except (NoSuchElementException, ElementNotInteractableException) as e:
870            print(f"Failed to perform zoom {direction} action: {str(e)}")
